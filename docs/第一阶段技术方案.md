# 末日废墟 - 第一阶段技术方案

> 版本：v1.0
> 日期：2025年
> 阶段目标：完成回合制卡牌核心玩法原型，单平台验证

---

## 一、技术栈确定

| 层级 | 技术选型 | 说明 |
|------|----------|------|
| 游戏引擎 | Cocos Creator 3.8.x | 支持6端发布，TypeScript开发 |
| 后端框架 | Node.js 20 + Colyseus | 实时对战框架，房间管理 |
| 数据库 | MongoDB + Redis | MongoDB存档，Redis缓存/会话 |
| 认证系统 | JWT | 跨平台用户认证 |
| 管理后台 | Vue3 + Element Plus | 可视化配置系统 |
| 部署方案 | Docker + 云服务器 | 灵活扩展 |

## 三、数据结构设计

### 3.1 英雄数据结构

```typescript
// 英雄系别枚举
enum HeroFaction {
  RED = 'red',      // 红系
  BLUE = 'blue',    // 蓝系
  GREEN = 'green',  // 绿系
  YELLOW = 'yellow', // 黄系
  LIGHT = 'light',  // 光系（特殊）
  DARK = 'dark',    // 暗系（特殊）
  GOD = 'god'       // 神系（特殊）
}

// 英雄职业枚举
enum HeroClass {
  ASSASSIN = 'assassin',  // 刺客
  MAGE = 'mage',          // 法师
  TANK = 'tank',          // 坦克
  WARRIOR = 'warrior',    // 战士
  HEALER = 'healer'       // 奶妈
}

// 英雄品质枚举
enum HeroQuality {
  THREE_STAR = 3,   // 三星
  FOUR_STAR = 4,    // 四星
  FIVE_STAR = 5     // 五星
}

// 英雄配置表结构
interface HeroConfig {
  id: string;                    // 英雄ID
  name: string;                  // 英雄名称
  faction: HeroFaction;          // 系别
  heroClass: HeroClass;          // 职业
  quality: HeroQuality;          // 品质

  // 基础属性
  baseStats: {
    hp: number;           // 生命值
    atk: number;          // 攻击力
    def: number;          // 防御力
    spd: number;          // 速度
    critRate: number;     // 暴击率
    critDamage: number;   // 暴击伤害
    armorPen: number;     // 破甲
  };

  // 成长系数
  growth: {
    hp: number;
    atk: number;
    def: number;
    spd: number;
  };

  // 技能列表
  skills: string[];              // 技能ID数组

  // 外观资源
  assets: {
    model: string;        // 3D模型路径
    icon: string;         // 头像图标
    portrait: string;     // 立绘
  };

  // 获取方式
  obtain: {
    fragments: number;    // 召唤所需碎片数
    sources: string[];    // 获取来源
  };
}

// 玩家英雄实例
interface PlayerHero {
  instanceId: string;            // 实例唯一ID
  heroId: string;                // 英雄配置ID
  level: number;                 // 等级
  star: number;                  // 星级（突破）

  // 计算后的属性
  stats: {
    hp: number;
    atk: number;
    def: number;
    spd: number;
    critRate: number;
    critDamage: number;
    armorPen: number;
  };

  // 装备槽
  equipment: {
    weapon: string | null;
    armor: string | null;
    accessory: string | null;
  };

  // 技能等级
  skillLevels: number[];

  // 经验值
  exp: number;

  // 创建时间
  createdAt: Date;
}
```

### 3.2 技能数据结构（配置化核心）

```typescript
// 技能类型枚举
enum SkillType {
  DAMAGE = 'damage',           // 伤害类
  HEAL = 'heal',               // 治疗类
  BUFF = 'buff',               // 增益类
  DEBUFF = 'debuff',           // 减益类
  CONTROL = 'control',         // 控制类
  SUMMON = 'summon'            // 召唤类
}

// 目标类型枚举
enum TargetType {
  SELF = 'self',                       // 自身
  ENEMY_SINGLE = 'enemy_single',       // 敌方单体
  ENEMY_ALL = 'enemy_all',             // 敌方全体
  ENEMY_RANDOM = 'enemy_random',       // 敌方随机
  ALLY_SINGLE = 'ally_single',         // 友方单体
  ALLY_ALL = 'ally_all',               // 友方全体
  ALLY_LOWEST_HP = 'ally_lowest_hp',   // 血量最低友方
  ALL = 'all'                          // 全体
}

// 控制效果枚举
enum ControlType {
  STUN = 'stun',         // 晕眩
  SILENCE = 'silence',   // 沉默
  FREEZE = 'freeze',     // 冰冻
  POISON = 'poison',     // 中毒
  BURN = 'burn',         // 燃烧
  BLEED = 'bleed'        // 流血
}

// 技能效果
interface SkillEffect {
  type: SkillType;               // 效果类型
  target: TargetType;            // 目标类型

  // 伤害/治疗效果
  value?: {
    base: number;                // 基础值
    coefficient: number;         // 系数（乘以攻击力）
    formula?: string;            // 自定义公式
  };

  // 控制效果
  control?: {
    type: ControlType;
    duration: number;            // 持续回合数
    chance: number;              // 触发概率
  };

  // Buff/Debuff效果
  buff?: {
    stat: string;                // 属性名
    value: number;               // 增减值
    percent: number;             // 百分比增减
    duration: number;            // 持续回合
  };

  // 特殊效果
  special?: {
    type: string;                // 特殊效果类型
    params: any;                 // 参数
  };
}

// 技能配置表结构
interface SkillConfig {
  id: string;                    // 技能ID
  name: string;                  // 技能名称
  description: string;           // 技能描述
  type: SkillType;               // 技能类型

  // 冷却配置
  cooldown: number;              // 冷却回合数
  initialCooldown: number;       // 初始冷却

  // 效果列表（按顺序执行）
  effects: SkillEffect[];

  // 动画配置
  animation: {
    cast: string;                // 施法动画
    hit: string;                 // 命中特效
    sound: string;               // 音效
  };

  // 触发条件（可选）
  condition?: {
    type: string;                // 条件类型
    params: any;                 // 条件参数
  };
}
```

### 3.3 战斗数据结构

```typescript
// 战斗单位（战斗中的英雄快照）
interface BattleUnit {
  instanceId: string;            // 实例ID
  heroId: string;                // 英雄配置ID
  team: 'attacker' | 'defender'; // 阵营

  // 位置（1-6）
  position: number;

  // 当前属性（含装备加成）
  currentStats: {
    hp: number;
    maxHp: number;
    atk: number;
    def: number;
    spd: number;
    critRate: number;
    critDamage: number;
    armorPen: number;
  };

  // 技能状态
  skills: {
    id: string;
    currentCooldown: number;
    level: number;
  }[];

  // 状态效果
  statusEffects: {
    type: ControlType;
    duration: number;
    value?: number;
    source: string;
  }[];

  // 是否存活
  isAlive: boolean;
}

// 战斗回合
interface BattleTurn {
  turnNumber: number;            // 回合数
  actions: BattleAction[];       // 行动列表
}

// 战斗行动
interface BattleAction {
  actorId: string;               // 行动者ID
  skillId: string;               // 使用的技能ID
  targets: string[];             // 目标ID列表

  // 行动结果
  results: {
    targetId: string;
    damage?: number;
    heal?: number;
    effects?: string[];
    isCritical?: boolean;
    isDodged?: boolean;
  }[];

  // 是否跳过（晕眩等）
  skipped?: boolean;
  skipReason?: string;
}

// 战斗结果
interface BattleResult {
  battleId: string;              // 战斗ID
  winner: 'attacker' | 'defender' | 'draw';
  totalTurns: number;

  // 战斗日志（用于回放）
  turns: BattleTurn[];

  // 奖励
  rewards?: {
    exp: number;
    gold: number;
    items: { id: string; count: number }[];
  };

  // 战斗时间
  startTime: Date;
  endTime: Date;
}
```

// 活动类型枚举
enum ActivityType {
  BEER = 'beer',           // 啤酒活动

  BLACK_WATER = 'black_water', // 黑水活动
  MONSTER = 'monster'      // 打怪兽活动

interface BeerActivityConfig {
  id: string;
  type: ActivityType.BEER;
  startTime: Date;
  endTime: Date;

  // 特殊系配置（光/暗/神）
  isSpecialFaction: boolean;

  rewards: {
    [points: number]: {
      fragments: {
        heroId?: string;        // 指定英雄碎片（管理后台设置）

      items: { id: string; count: number }[];
    };
  };

  // 做满一轮奖励
  fullRoundReward: {
    fragments: { heroId?: string; count: number };
    monthlyCoins: number;
  };

  // 特殊系满轮额外奖励
  specialExtraReward?: {
    fragments: { heroId: string; count: number };
  };
}

// 活动周期配置
interface ActivityCycleConfig {
  cycleLength: number;           // 周期长度（4周）
  activities: {
    week: number;
    type: ActivityType;

  }[];

## 九、风险与应对

| 风险 | 影响 | 应对策略 |
|------|------|----------|
| 3D资源质量不达标 | 中 | 先购买低成本资源验证，后续升级 |
| 战斗系统复杂度高 | 高 | 先实现基础版本，逐步迭代 |
| 性能问题（多单位战斗） | 中 | 优化DrawCall、使用对象池 |
| 微信小程序包体限制 | 高 | 资源远程加载、分包策略 |
| 实时同步延迟 | 中 | 使用权威服务器+客户端预测 |
| 个人精力不足 | 高 | 严格控制范围，延期MOBA模块 |
